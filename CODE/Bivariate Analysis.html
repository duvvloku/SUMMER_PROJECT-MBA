import scipy.stats as stats

f_stat, p_value = stats.f_oneway(data['V1'][data['style'] == 'V'],
               data['V1'][data['style'] == 'A'],
               data['V1'][data['style'] == 'R'],
               data['V1'][data['style'] == 'K'],
               data['V1'][data['style'] == 'Multi'])

Ho = "V1 has no effect on learning_style"   # Stating the Null Hypothesis
Ha = "V1 has an effect on learning_style"   # Stating the Alternate Hypothesis

print('f_statistic : ',f_stat)
print('')
print('p_value : ',p_value)
print('')

if p_value < 0.05:  # Setting our significance level at 5%
    print(f'{Ha} as the p_value ({p_value.round(3)}) < 0.05')
else:
    print(f'{Ho} as the p_value ({p_value.round(3)}) > 0.05')
